function solution(k, m, score) {
    let answer = 0;
    score.sort((a, b) => b - a);
    for (let i = 0; i < parseInt(score.length / m); i++) {
        const box = score.slice(i * m, i * m + m);
        const minScoreInBox = box[m - 1];
        answer += minScoreInBox * m;
    }
    return answer;
}
