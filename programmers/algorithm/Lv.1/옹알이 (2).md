function solution(babbling) {
  const sounds = ["aya", "ye", "woo", "ma"];
  let count = 0;

  babbling.forEach((word) => {
    let tmp = word;
    let lastSound = "";
    let isMatched = true;

    while (tmp.length > 0) {
      let matched = false;

      for (let sound of sounds) {
        if (tmp.startsWith(sound) && sound !== lastSound) {
          tmp = tmp.slice(sound.length);
          lastSound = sound;
          matched = true;
          break;
        }
      }

      if (!matched) {
        isMatched = false;
        break;
      }
    }

    if (isMatched) count++;
  });
  return count;
}
