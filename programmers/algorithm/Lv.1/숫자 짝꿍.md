function solution(X, Y) {
  let answer = "";
  X = X.split("").sort((a, b) => b - a);
  Y = Y.split("").sort((a, b) => b - a);
  let p1 = 0;
  let p2 = 0;
  while (p1 < X.length && p2 < Y.length) {
    if (X[p1] === Y[p2]) {
      answer += X[p1++];
      p2++;
    } else if (X[p1] > Y[p2]) p1++;
    else p2++;
  }
  if (!answer.length) return "-1";
  else if (!Number(answer)) return "0";
  return answer;
}
